Datatypes Examples

- Get the name and extension of "salesforce.com". 

         String str = 'salesforce.com'; 
         
         String str1 = str.substring(0, str.indexOf('.'));
         System.debug(str1);  // salesforce 
         String str2 = str.substring(str.indexOf('.')+1);
         System.debug(str2);  // com 

          // Another way 
         String str1 = str.substringAfter('.'); 
         System.debug(str1);  // com  
         String str2 = str.substringBefore('.'); 
         System.debug(str2);  // salesforce 
         
          // Another way 
         List strArray = str.split('\\.');
         System.debug(strArray[0]);  // salesforce 
         System.debug(strArray[1]);  // com

- Take the words "World" and "Everybody" in the phrase "Hello World Everybody". 

         String str = 'Hello World Everbody';
         String str1 = str.substringAfterLast(' ');
         System.debug(str1);  // Everbody
         String str2 = str.substringBetween(' ');
         System.debug(str2);  // World

- Convert a value coming from an external system in "apple, pear, cherry" format to multiselect picklist format. 

         String str = 'apple, pear, cherry';
         str = str.replaceAll(',', ';');
         System.debug(str);  // apple; pear; cherry

- Create two dates and find the number of months, days and years between them. 

         Date firstDate = Date.newInstance(2002, 5, 2);
         Date secondDate = Date.newInstance(2006, 6, 10);
         Integer monthsBetween = firstDate.monthsBetween(secondDate);
         System.debug(monthsBetween);  // 49 

         Integer daysBetween = firstDate.daysBetween(secondDate);
         System.debug(daysBetween);  // 1500 
 
         System.debug(Math.abs(firstDate.year() - secondDate.year()));  // 4

- Convert a string word to a list and then restore it. 

         String str = 'Hello';
         List liste = str.split('');
         System.debug(liste);  // (H,e,l,l,o)

         string strAgain = liste.toString();
         strAgain = strAgain.substringBetween('(',')').remove(',').deleteWhiteSpace();
         System.debug(strAgain);  // Hello
         
          // Another way
         String text = '';
         for(string w:liste){
            text += w;
         }    
         System.debug(strAgain);  // Hello

- Convert a string sentence to a list and then undo it again. 

         String text = 'Bu bir cümledir.';
         List textList = text.remove('.').split(' ');
         System.debug(textList);  // (Bu, bir, cümledir)
         
         String strAgain = textList.toString();
         strAgain = strAgain.substringBetween('(',')').remove(',') + '.';
         System.debug(strAgain);  // Bu bir cümledir.
         
          // Another way
         String text = '';
         for(String w:textList){
            text += w + ' ';
         }
         text = text.trim() + '.';
         System.debug(text);  // (Bu, bir, cümledir)
